    داده‌ها فقط روی همین آیفون ذخیره می‌شوند (LocalStorage).<br>

  <script>
    // ثبت Service Worker برای حالت آفلاین
    if ('serviceWorker' in navigator) {
      window.addEventListener('load', () => {
        navigator.serviceWorker.register('./service-worker.js');
      });
    }

    const input = document.getElementById('ticketInput');
    const addBtn = document.getElementById('addTicketBtn');
    const list = document.getElementById('ticketList');

    function loadTickets() {
      const raw = localStorage.getItem('tickets');
      let tickets = [];
      try {
        tickets = raw ? JSON.parse(raw) : [];
      } catch (e) {
        tickets = [];
      }
      renderTickets(tickets);
    }

    function saveTickets(tickets) {
      localStorage.setItem('tickets', JSON.stringify(tickets));
    }

    function renderTickets(tickets) {
      if (!tickets.length) {
        list.innerHTML = '<div class="empty">هنوز هیچ بلیتی ذخیره نکرده‌ای.</div>';
        return;
      }
      list.innerHTML = '';
      tickets.forEach((t, index) => {
        const div = document.createElement('div');
        div.className = 'ticket';
        div.innerHTML = `
          <span>${t}</span>
          <button class="btn" style="background:#ef4444;font-size:11px;padding:6px 10px;" data-index="${index}">
            حذف
          </button>
        `;
        list.appendChild(div);
      });

      list.querySelectorAll('button[data-index]').forEach(btn => {
        btn.addEventListener('click', (e) => {
          const idx = parseInt(e.target.getAttribute('data-index'), 10);
          const raw = localStorage.getItem('tickets');
          let tickets = raw ? JSON.parse(raw) : [];
          tickets.splice(idx, 1);
          saveTickets(tickets);
          renderTickets(tickets);
        });
      });
    }

    addBtn.addEventListener('click', () => {
      const value = input.value.trim();
      if (!value) return;
      const raw = localStorage.getItem('tickets');
      let tickets = raw ? JSON.parse(raw) : [];
      tickets.push(value);
      saveTickets(tickets);
      input.value = '';
      renderTickets(tickets);
    });

    loadTickets();
  </script>
